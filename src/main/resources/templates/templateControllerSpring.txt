package PACKAGE_ROOT.controller;

import java.util.ArrayList;
import java.util.List;

import org.springframework.web.bind.annotation.ModelAttribute;
import org.springframework.web.bind.annotation.PathVariable;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RequestMethod;
import org.springframework.web.bind.annotation.RestController;

import PACKAGE_POJOS.dao.CLASSDao;
import PACKAGE_POJOS.daoImpl.CLASSDaoImpl;
import PACKAGE_POJOS.CLASS;

@RestController
public class CLASSController {

	private CLASSDao dao = new CLASSDaoImpl();

	@RequestMapping(value = "/CLASSs", method = RequestMethod.GET, headers = "Accept=application/json")
	public List<CLASS> getAll() {
		List<CLASS> lista = new ArrayList<CLASS>();
		try {
			lista = dao.getAll();
		} catch (Exception e) {
			e.printStackTrace();
		}
		return lista;
	}

	@RequestMapping(value = "/CLASS/{id}", method = RequestMethod.GET, headers = "Accept=application/json")
	public CLASS getByIdCLASS(@PathVariable int id) {
		CLASS OBJECT = new CLASS();
		OBJECT.setIdCLASS(id);
		try {
			OBJECT = dao.getById(id);
		} catch (Exception e) {
			e.printStackTrace();
		}
		return OBJECT;
	}

	@RequestMapping(value = "/CLASS", method = RequestMethod.POST, headers = "Accept=application/json")
	public CLASS saveCLASS(@ModelAttribute CLASS OBJECT) {
		try {	
			dao.save(OBJECT);
		} catch (Exception e) {
			e.printStackTrace();
		}
		return OBJECT;
	}

	@RequestMapping(value = "/CLASS", method = RequestMethod.PUT, headers = "Accept=application/json")
	public CLASS updateCLASS(@ModelAttribute CLASS OBJECT) {
		try {
			dao.update(OBJECT);
		} catch (Exception e) {
			e.printStackTrace();
		}
		return OBJECT;

	}

	@RequestMapping(value = "/CLASS/{id}", method = RequestMethod.DELETE, headers = "Accept=application/json")
	public void deleteCLASS(@PathVariable("id") int id) {
		CLASS OBJECT = new CLASS();
		OBJECT.setIdCLASS(id);
		try {
			OBJECT = dao.getById(id);
			dao.delete(OBJECT);
		} catch (Exception e) {
			e.printStackTrace();
		}
	}
}
